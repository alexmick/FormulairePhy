\indent $\circledast$ Miroirs sphériques:
\[
\gamma=\frac{\overline{FA'}}{\overline{FS}}=\frac{\overline{FS}}{\overline{FA}}
=-\frac{\overline{SA'}}{\overline{SA}}=\frac{\overline{CA'}}{\overline{CA}} \hspace{10mm}
\overline{FA}\:\overline{FA'}=FS^2\ (\textsc{Newton}) \hspace{10mm}
\frac{1}{\overline{SA'}}+\frac{1}{\overline{SA}}=\frac{1}{\overline{SF}}\ (\textsc{Descartes})
\]
\indent $\circledast$ Lentilles:
\[
\gamma=\frac{\overline{F'A'}}{\overline{F'O}}=\frac{\overline{FO}}{\overline{FA}}=\frac{\overline{OA'}}{\overline{OA}} \hspace{10mm}
\overline{FA}\:\overline{FA'}=-f'^2\ (\textsc{Newton}) \hspace{10mm}
\frac{1}{\overline{OA'}}-\frac{1}{\overline{OA}}=\frac{1}{f'}\ (\textsc{Descartes})
\]
